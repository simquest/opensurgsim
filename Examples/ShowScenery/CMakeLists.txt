link_directories(
	${Boost_LIBRARY_DIRS}
)

	"${OSG_INCLUDE_DIR}"
	"${OPENTHREADS_INCLUDE_DIR}"
)

	ShowScenery.cpp
)

)

	SurgSimBlocks
	SurgSimGraphics
	${Boost_LIBRARIES}
	${YAML_CPP_LIBRARIES}
)

# Put ShowScenery into folder "Examples"
set_target_properties(ShowScenery PROPERTIES FOLDER "Examples")
